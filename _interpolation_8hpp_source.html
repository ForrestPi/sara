33  // Interpolation
35  template <typename T, int N>
36  typename PixelTraits<T>::template Cast<double>::pixel_type
37  interpolate(const Image<T, N>& image, const Matrix<double, N, 1>& pos)
38  {
39  // Typedefs.
40  typedef typename PixelTraits<T>::template Cast<double>::pixel_type
41  DoublePixel;
42  typedef typename Image<T, N>::const_subarray_iterator
43  const_subarray_iterator;
44 
45  // Find the smallest integral bounding box that encloses the position.
46  Matrix<int, N, 1> start, end;
47  Matrix<double, N, 1> frac;
48  for (int i = 0; i < N; ++i)
49  {
50  if (pos[i] < 0 || pos[i] >= image.size(i))
51  throw std::range_error("Cannot interpolate: position is out of range");
52 
53  double ith_int_part;
54  frac[i] = std::modf(pos[i], &ith_int_part);
55  start[i] = static_cast<int>(ith_int_part);
56  }
57  end.array() = start.array() + 2;
58 
59  // Compute the weighted sum.
60  const_subarray_iterator it(image.begin_subarray(start, end));
61  DoublePixel interpolated_value(PixelTraits<DoublePixel>::min());
62  Matrix<int, N, 1> offset;
63  for ( ; !it.end(); ++it)
64  {
65  double weight = 1.;
66  for (int i = 0; i < N; ++i)
67  {
68  weight *= (it.position()[i] == start[i]) ? (1.-frac[i]) : frac[i];
69  offset[i] = it.position()[i] < image.size(i) ? 0 : -1;
70  }
71 
72  DoublePixel dst_color;
73  dst_color = PixelTraits<T>::template Cast<double>::apply(it(offset));
74  interpolated_value += weight*dst_color;
75  }
76  return interpolated_value;
77  }
