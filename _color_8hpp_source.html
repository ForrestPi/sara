357  template <typename T>
358  inline double getRescaledChannel64f(T value)
359  {
360  return (static_cast<double>(value) - ChannelTraits<T>::doubleMin())
361  / ChannelTraits<T>::doubleRange();
362  }
364  template <typename T, int N>
365  inline Matrix<double, N, 1> getRescaledColor64f(const Matrix<T, N, 1>& color)
366  {
367  Matrix<double, N, 1> min; min.fill(ChannelTraits<T>::doubleMin());
368  Matrix<double, N, 1> max; max.fill(ChannelTraits<T>::doubleMax());
369  return (color.template cast<double>() - min).cwiseQuotient(max - min);
370  }
372  template <typename T>
373  inline void normalizeChannel(T& dst, double src)
374  {
375  dst = static_cast<T>(
376  ColorTraits<T>::doubleMin() + src*ColorTraits<T>::doubleRange() );
377  }
379  template <typename T, int N>
380  inline void normalizeColor(Matrix<T, N, 1>& dst, const Matrix<double, N, 1>& src)
381  {
382  typedef Color<T, Rgb> Col;
383  Matrix<double, N, 1> tmp( ColorTraits<Col>::doubleMin()
384  + src.cwiseProduct(ColorTraits<Col>::doubleRange()) );
385  dst = tmp.template cast<T>();
386  }

390  template <typename T>
391  inline double rgb2gray64f(const Matrix<T, 3, 1>& rgb)
392  {
393  Matrix<double, 3, 1> rgb64f(getRescaledColor64f(rgb));
394  return 0.3*rgb64f[0] + 0.59*rgb64f[1] + 0.11*rgb64f[2];
395  }
397  template <typename T>
398  inline Matrix<double, 3, 1> gray2rgb64f(T gray)
399  {
400  double gray64f = getRescaledChannel64f(gray);
401  Matrix<double, 3, 1> rgb64f;
402  rgb64f.fill(gray64f);
403  return rgb64f;
404  }
406  template<typename T>
407  inline Vector3d rgb2yuv64f(const Matrix<T, 3, 1>& rgb)
408  {
409  Matrix<double, 3, 1> rgb64f(getRescaledColor64f(rgb));
410 
411  double y = .299*rgb64f[0] + .587*rgb64f[1] + .114*rgb64f[2];
412  return Vector3d(y, .492*(rgb64f[2]-y), .877*(rgb64f[0]-y));
413  }
415  template<typename T>
416  inline Vector3d yuv2rgb64f(const Matrix<T, 3, 1>& yuv)
417  {
418  Vector3d yuv64f(getRescaledColor64f(yuv));
419  double r = yuv64f[2]/.877 + yuv64f[0];
420  double b = yuv64f[1]/.492 + yuv64f[0];
421  double g = (yuv64f[0] - .299*r - .114*b) / .587;
422  return Matrix<double, 3, 1>(r, g, b);
423  }
