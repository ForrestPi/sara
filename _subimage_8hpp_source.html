20  template <typename T, int N>
21  Image<T, N> getImagePatch(const Image<T, N>& src,
22  const Matrix<int, N, 1>& a,
23  const Matrix<int, N, 1>& b)
24  {
25  Image<T,N> dst(b-a);
26  dst.array().fill(ColorTraits<T>::zero());
27  CoordsIterator<N> c(a,b), end;
28  for (typename Image<T, N>::iterator dst_it = dst.begin();
29  dst_it != dst.end(); ++dst_it, ++c)
30  {
31  if ((*c-a).minCoeff() < 0 || (b-*c).minCoeff() >= 0)
32  continue;
33  *dst_it = src(*++c);
34  }
35  return dst;
36  }

39  template <typename T>
40  Image<T> getImagePatch(const Image<T>& src, int x, int y, int w, int h)
41  {
42  Image<T> patch(w,h);
43  patch.array().fill(ColorTraits<T>::zero());
44  for (int v = 0; v < h; ++v)
45  {
46  if (y+v < 0 || y+v > src.height()-1)
47  continue;
48  for (int u = 0; u < w; ++u)
49  {
50  if ( x+u < 0 || x+u > src.width()-1 )
51  continue;
52  patch(u,v) = src(x+u,y+v);
53  }
54  }
55  return patch;
56  }
