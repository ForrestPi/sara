72  template <typename T, int N>
73  void computeOrientationHistogram(Array<T, N, 1>& oriHist,
74  const Image<Matrix<T,2,1> >& gradPolar,
75  T x, T y, T s,
76  int patchTruncationFactor = 3,
77  T blurFactor = T(1.5))
78  {
79  // Weighted histogram of gradients.
80  oriHist.setZero();
81 
82  // Rounding of the coordinates.
83  int xi = intRound(x);
84  int yi = intRound(y);
85 
86  // std deviation of the gaussian weight (cf. [Lowe, IJCV 2004])
87  T sigma = s*blurFactor;
88 
89  // Patch radius on which the histogram of gradients is performed.
90  int patchRadius = intRound(sigma*patchTruncationFactor);
91 
92  // Accumulate the histogram of orientations.
93  for (int v = -patchRadius; v <= patchRadius; ++v)
94  {
95  for (int u = -patchRadius; u <= patchRadius; ++u)
96  {
97  if ( xi+u < 0 || xi+u >= gradPolar.width() ||
98  yi+v < 0 || yi+v >= gradPolar.height() )
99  continue;
100 
101  T mag = gradPolar(xi+u, yi+v)(0);
102  T ori = gradPolar(xi+u, yi+v)(1);
103  // ori is in \f$]-\pi, \pi]\f$, so translate ori by \f$2*\pi\f$ if it is
104  // negative.
105 #ifndef LOWE
106  ori = ori < 0 ? ori+T(2.*M_PI) : ori;
107  int binIndex = floor(ori/T(2*M_PI) * T(N));
108  binIndex %= N;
109 #else
110  int binIndex = int( (N * (ori + M_PI + 0.001f) / (2.0f * M_PI)) );
111  binIndex = std::min(binIndex, N - 1);
112 #endif
113  if (binIndex < 0 || binIndex >= N)
114  {
115  std::ostringstream oss;
116  oss << "Orientation bin index out of range: " << binIndex
117  << " theta = " << ori << std::endl;
118  std::cerr << oss.str() << std::endl;
119  throw std::out_of_range(oss.str());
120  }
121 
122 
123  // Give more emphasis to gradient orientations that lie closer to the
124  // keypoint location.
125  T weight = exp(-(u*u+v*v)/(T(2)*sigma*sigma));
126  // Also give more emphasis to gradient with large magnitude.
127  oriHist( binIndex ) += weight*mag;
128  }
129  }
130  }

163  template <typename T, int N>
164  std::vector<int> findPeaks(const Array<T, N, 1>& oriHist,
165  T peakRatioThres = T(0.8))
166  {
167  T max = oriHist.maxCoeff();
168  std::vector<int> oriPeaks;
169  oriPeaks.reserve(N);
170  for (int i = 0; i < N; ++i)
171  if ( oriHist(i) >= peakRatioThres*max &&
172  oriHist(i) > oriHist((i-1+N)%N) &&
173  oriHist(i) > oriHist((i+1)%N) )
174  oriPeaks.push_back(i);
175  return oriPeaks;

181  template <typename T, int N>
182  T refinePeak(const Array<T, N, 1>& oriHist, int i)
183  {
184  T y0 = oriHist( (i-1+N)%N );
185  T y1 = oriHist( i );
186  T y2 = oriHist( (i+1)%N );
187  // Denote the orientation histogram function by \f$f\f$.
188  // perform a 2nd-order Taylor approximation:
189  // \f$f(x+h) = f(x) + f'(x)h + f''(x) h^2/2\f$
190  // We approximate \f$f'\f$ and \f$f''\f$ by finite difference.
191  T fprime = (y2-y0) / 2.f;
192  T fsecond = y0 - 2.f*y1 + y2;
193  // Maximize w.r.t. to \f$h\f$, derive the expression.
194  // Thus \f$h = -f'(x)/f''(x)\f$.
195  T h = -fprime / fsecond;
196  // Add the offset \f$h\f$ to get the refined orientation value.
197  // Note that we also add the 0.5f offset, because samples are assumed taken
198  // on the middle of the interval \f$[i, i+1)\f$.
199  return T(i)+T(0.5)+h;
200  }
202  template <typename T, int N>
203  std::vector<T> refinePeaks(const Array<T, N, 1>& oriHist,
204  const std::vector<int>& ori)
205  {
206  std::vector<T> oriT(ori.size());
207  for (size_t i = 0; i < ori.size(); ++i)
208  oriT[i] = refinePeak<T,N>(oriHist, ori[i]);
209  return oriT;
210  }
