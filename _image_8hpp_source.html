213  // Image rescaling functions
215  template <typename T, typename Layout, int N>
216  inline Image<Color<T,Layout>, N> colorRescale(
217  const Image<Color<T,Layout>, N>& src,
218  const Color<T, Layout>& a = black<T>(),
219  const Color<T, Layout>& b = white<T>())
220  {
221  Image<Color<T,Layout>, N> dst(src.sizes());
222 
223  const Color<T,Layout> *src_first = src.data();
224  const Color<T,Layout> *src_last = src_first + src.size();
225  Color<T,Layout> *dst_first = dst.data();
226 
227  Color<T,Layout> min(*src_first);
228  Color<T,Layout> max(*src_first);
229  for ( ; src_first != src_last; ++src_first)
230  {
231  min = min.cwiseMin(*src_first);
232  max = max.cwiseMax(*src_first);
233  }
234 
235  if (min == max)
236  {
237  std::cerr << "Warning: min == max!" << std::endl;
238  return dst;
239  }
240 
241  for (src_first = src.data(); src_first != src_last;
242  ++src_first, ++dst_first)
243  *dst_first = a + (*src_first-min).cwiseProduct(b-a).
244  cwiseQuotient(max-min);
245 
246  return dst;
247  }
