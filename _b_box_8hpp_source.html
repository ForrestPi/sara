31  BBox(const Point2d& top_left, const Point2d& bottom_right)

46  BBox(const Point2d *begin, const Point2d *end)
47  {
48  if (!begin)
49  {
50  const char *msg = "The array of points seems wrong.";
51  throw std::logic_error(msg);
52  }
53 
54  _top_left = _bottom_right = *begin;
55  for (const Point2d *p = begin; p != end; ++p)
56  {
57  _top_left.x() = std::min(_top_left.x(), p->x());
58  _top_left.y() = std::min(_top_left.y(), p->y());
59  _bottom_right.x() = std::max(_bottom_right.x(), p->x());
60  _bottom_right.y() = std::max(_bottom_right.y(), p->y());
61  }
62  }
