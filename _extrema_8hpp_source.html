26  struct CompareWithNeighborhood3
27  {
28  bool operator()(T val, int x, int y, const Image<T>& I,
29  bool compareWithCenter) const
30  {
31  for (int v=-1; v <= 1; ++v)
32  {
33  for (int u=-1; u <= 1; ++u)

36  continue;

39  }
40  }

42  }
