20  template <int N, int Axis>
21  struct Differential
22  {
23  DO_STATIC_ASSERT(Axis >= 0 && Axis < N,
24  AXIS_MUST_NONNEGATIVE_AND_LESS_THAN_N);

26  template <typename T, int StorageOrder>
27  static inline T eval_derivative(Locator<T, N, StorageOrder>& loc)
28  {
29  return (loc.template axis<Axis>()[1] - loc.template axis<Axis>()[-1])
30  / static_cast<T>(2);
31  }
32 
33  template <typename T, int StorageOrder>
34  static inline T accumulate_neighbor_values(Locator<T, N, StorageOrder>& loc)
35  {
36  return loc.template axis<Axis>()[1]+loc.template axis<Axis>()[-1]
37  + Differential<N, Axis-1>::accumulate_neighbor_values(loc);
38  }

40  template <typename T, int StorageOrder>
41  static inline void eval_gradient(Matrix<T, N, 1>& g,
42  Locator<const T, N, StorageOrder>& loc)
43  {
44  g[Axis] = eval_derivative(loc);
45  Differential<N, Axis-1>::eval_gradient(g, loc);
46  }
47 
48  template <typename T, int StorageOrder>
49  static inline T eval_laplacian(Locator<T, N, StorageOrder>& loc)
50  { return accumulate_neighbor_values(loc) - 2*N*(*loc); }
51 
52  template <typename T, int StorageOrder>
53  static inline T eval_divergence(Locator<Matrix<T, N, 1>, N, StorageOrder>& loc)
54  {
55  return eval_derivative(loc)
56  + Differential<N, Axis-1>::eval_divergence(loc);
57  }

161  template <typename T, int N = 2>
162  struct ComputeHessian
163  {
164  typedef Matrix<int, N, 1> Coords, Vector;
165  typedef Image<T, N> ScalarField;
166  typedef Matrix<T, N, N> Matrix;
167  typedef Image<Matrix, N> MatrixField;
168  typedef typename ScalarField::ConstLocator ScalarLocator;
169  typedef typename MatrixField::Locator MatrixLocator;
170 
171  inline ComputeHessian(const ScalarField& scalarField)
172  : scalar_field_(scalarField) {}
173 
174  void operator()(Matrix& H, const ScalarLocator& loc) const
175  {
176  /* Just a reminder:
177  *
178  * ixx = ( i(x+1,y) - 2*i(x,y) + i(x-1,y) ) / static_cast<T>(4);
179  * iyy = ( i(x,y+1) - 2*i(x,y) + i(x,y-1) ) / static_cast<T>(4);
180  * ixy = ( (i(x+1,y+1) - i(x-1,y+1)) - (i(x+1,y-1) - i(x-1,y-1)) )
181  * / static_cast<T>(4);
182  */
183  for (int i = 0; i < N; ++i)
184  for (int j = i; j < N; ++j)
185  {
186  if (i == j)
187  H(i,i) = ( loc(delta(i,1)) - 2*(*loc) + loc(delta(i,-1)) )
188  / static_cast<T>(4);
189  else
190  {
191  H(i,j) = ( ( loc(delta(i,1,j, 1)) - loc(delta(i,-1,j, 1)) )
192  - ( loc(delta(i,1,j,-1)) - loc(delta(i,-1,j,-1)) ) )
193  / static_cast<T>(4);
194  H(j,i) = H(i,j);
195  }
196  }
197  }
