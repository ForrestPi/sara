31  template <typename T>
33  ImagePyramid<T>
34  gaussian_pyramid(const Image<T>& image,
35  const ImagePyramidParams& params = ImagePyramidParams())
36  {
37  typedef typename ImagePyramid<T>::scalar_type Scalar;
38  // Resize the image with the appropriate factor.
39  Scalar resize_factor = pow(2.f, -params.first_octave_index());
40  Image<T> I(enlarge(image, resize_factor) );
41  // Deduce the new camera sigma with respect to the dilated image.
42  Scalar camera_sigma = Scalar(params.scale_camera())*resize_factor;
43  // Blur the image so that its new sigma is equal to the initial sigma.
44  Scalar init_sigma = Scalar(params.scale_initial());
45  if (camera_sigma < init_sigma)
46  {
47  Scalar sigma = sqrt(init_sigma*init_sigma - camera_sigma*camera_sigma);
48  I = gaussian(I, sigma);
49  }
50 
51  // Deduce the maximum number of octaves.
52  int l = std::min(image.width(), image.height());
53  int b = params.image_padding_size();
54  // l/2^k > 2b
55  // 2^k < l/(2b)
56  // k < log(l/(2b))/log(2)
57  int num_octaves = static_cast<int>(log(l/(2.f*b))/log(2.f));
58 
59  // Shorten names.
60  Scalar k = Scalar(params.scale_geometric_factor());
61  int num_scales = params.num_scales_per_octave();
62  int downscale_index = int( floor( log(Scalar(2))/log(k)) );
63 
64  // Create the image pyramid
65  ImagePyramid<T> G;
66  G.reset(num_octaves, num_scales, init_sigma, k);
67 
68  for (int o = 0; o < num_octaves; ++o)
69  {
70  // Compute the octave scaling factor
71  G.octave_scaling_factor(o) =
72  (o == 0) ? 1.f/resize_factor : G.octave_scaling_factor(o-1)*2;
73 
74  // Compute the gaussians in octave \f$o\f$
75  Scalar sigma_s_1 = init_sigma;
76  G(0, o) = o == 0 ? I : downscale(G(downscale_index, o - 1), 2);
77 
78  for (int s = 1; s < num_scales; ++s)
79  {
80  Scalar sigma = sqrt(k*k*sigma_s_1*sigma_s_1 - sigma_s_1*sigma_s_1);
81  G(s,o) = gaussian(G(s-1,o), sigma);
82  sigma_s_1 *= k;
83  }
84  }
85 
86  // Done!
87  return G;
88  }

91  template <typename T>
92  ImagePyramid<T> difference_of_gaussians_pyramid(const ImagePyramid<T>& gaussians)
93  {
94  ImagePyramid<T> D;
95  D.reset(gaussians.num_octaves(),
96  gaussians.num_scales_per_octave()-1,
97  gaussians.scale_initial(),
98  gaussians.scale_geometric_factor());
99 
100  for (int o = 0; o < D.num_octaves(); ++o)
101  {
102  D.octave_scaling_factor(o) = gaussians.octave_scaling_factor(o);
103  for (int s = 0; s < D.num_scales_per_octave(); ++s)
104  {
105  D(s,o).resize(gaussians(s,o).sizes());
106  D(s,o).array() = gaussians(s+1,o).array() - gaussians(s,o).array();
107  }
108  }
109  return D;
110  }

126  for (int s = 0; s < LoG.num_scales_per_octave(); ++s)

128  LoG(s,o) = laplacian(gaussians(s,o));

131  }
132  }

139  template <typename T>
140  Matrix<T, 3, 1> gradient(const ImagePyramid<T>& I, int x, int y, int s, int o)
141  {
142  // Sanity check
143  if (x < 1 || x >= I(s,o).width()-1 ||
144  y < 1 || y >= I(s,o).height()-1 ||
145  s < 1 || s >= static_cast<int>(I(o).size())-1)
146  {
147  std::ostringstream msg;
148  msg << "Fatal error: gradient out of range!" << std::endl;
149  std::cerr << msg.str() << std::endl;
150  throw std::out_of_range(msg.str());
151  }
152 
153  Matrix<T, 3, 1> d;
154  d(0) = (I(x+1,y ,s ,o) - I(x-1,y ,s ,o)) / T(2);
155  d(1) = (I(x ,y+1,s ,o) - I(x ,y-1,s ,o)) / T(2);
156  d(2) = (I(x ,y ,s+1,o) - I(x ,y ,s-1,o)) / T(2);
157  return d;
158  }

163  template <typename T>
164  Matrix<T, 3, 3> hessian(const ImagePyramid<T>& I, int x, int y, int s, int o)
165  {
166  // Sanity check
167  if (x < 1 || x >= I(s,o).width()-1 ||
168  y < 1 || y >= I(s,o).height()-1 ||
169  s < 1 || s >= static_cast<int>(I(o).size())-1)
170  {
171  std::string msg("Fatal error: Hessian out of range!");
172  std::cerr << msg << std::endl;
173  throw std::out_of_range(msg);
174  }
175 
176  Matrix<T, 3, 3> H;
177  // Ixx
178  H(0,0) = I(x+1,y,s,o) - T(2)*I(x,y,s,o) + I(x-1,y,s,o);
179  // Iyy
180  H(1,1) = I(x,y+1,s,o) - T(2)*I(x,y,s,o) + I(x,y-1,s,o);
181  // Iss
182  H(2,2) = I(x,y,s+1,o) - T(2)*I(x,y,s,o) + I(x,y,s-1,o);
183  // Ixy = Iyx
184  H(0,1) = H(1,0) = ( I(x+1,y+1,s,o) - I(x-1,y+1,s,o)
185  - I(x+1,y-1,s,o) + I(x-1,y-1,s,o) ) / T(4);
186  // Ixs = Isx
187  H(0,2) = H(2,0) = ( I(x+1,y,s+1,o) - I(x-1,y,s+1,o)
188  - I(x+1,y,s-1,o) + I(x-1,y,s-1,o) ) / T(4);
189  // Iys = Isy
190  H(1,2) = H(2,1) = ( I(x,y+1,s+1,o) - I(x,y-1,s+1,o)
191  - I(x,y+1,s-1,o) + I(x,y-1,s-1,o) ) / T(4);
192  // Done!
193  return H;
194  }
