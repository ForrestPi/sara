55  descriptor_type operator()(float x, float y, float sigma, float theta,
56  const Image<Vector2f>& grad_polar_coords) const
57  {
58  const float pi = static_cast<float>(M_PI);
59  /*
60  The oriented keypoint is denoted by $k = (x,y,\sigma,\theta)$.
61  SIFT describes keypoint $k$ in a similarity-invariant manner.
62 
63  To do so, we consider a square image patch which:
64  - is centered in $(x,y)$
65  - has an orientation angle $\theta$ w.r.t. the image frame coordinates:
66  => to ensure rotation invariance
67  - has a side length proportional to the scale $\sigma$:
68  => to ensure scale invariance
69  This square patch is denoted by $P(x,y,\sigma,\theta) = P(k)$.
70 
71  The square patch $P(x,y,\sigma,\theta)$ is itself divided into NxN
72  smaller square patches $(P_{i,j})_{1 \leq i \leq N, j \leq j \leq N}$.
73 
74  Notice that we omit the variables $(x,y,\sigma,\theta)$ which the
75  patches $P_{i,j}$ actually depend on.
76 
77  $N$ corresponds to the template argument 'int N' which should be 4 as
78  stated in the paper [Lowe, IJCV 2004]).
79 
80  In the image, each small square patch $P_{i,j}$ has a side length $l$
81  proportional to the scale $\sigma$ of the keypoint, i.e.,
82  $l = \lambda \sigma$.
83  */
84  const float lambda = _bin_scale_unit_length;
85  const float l = lambda*sigma;
86  /*
87  It is important to note that $\lambda$ is some 'universal' constant
88  used for all SIFT descriptors to ensure the scale-invariance of the
89  descriptor.
90  */
91 
92  /*
93  Now in each image square patch $P_{i,j}$, we build a histogram of
94  gradient orientations $\mathbf{h}_{i,j} \in \mathbb{R}^d$, which
95  quantizes the gradient orientations into $O$ principal orientations.
96  $O$ corresponds to the template argument 'int O'.
97 
98  Let us initialize the SIFT descriptor consisting of the NxN histograms
99  $\mathbf{h}_{i,j}$, each in $\mathbf{R}^O$ as follows.
100  */
101  descriptor_type h{ descriptor_type::Zero() };
102 
103  /*
104  In the rescaled and oriented coordinate frame bound to the patch $P(k)$,
105  - keypoint $k$ is located at (0,0)
106  - centers $C_{i,j}$ of patch $P_{i,j}$ are located at
107  $[ -(N+1)/2 + i, -(N+1)/2 + j ]$
108 
109  For example for $N=4$, they are at:
110  (-1.5,-1.5) (-0.5,-1.5) (0.5,-1.5) (1.5,-1.5)
111  (-1.5,-0.5) (-0.5,-0.5) (0.5,-0.5) (1.5,-0.5)
112  (-1.5, 0.5) (-0.5, 0.5) (0.5, 0.5) (1.5, 0.5)
113  (-1.5, 1.5) (-0.5, 1.5) (0.5, 1.5) (1.5, 1.5)
114 
115  Gradients in $[x_i-1, x_i+1] \times [y_i-1, y_i+1]$ contributes
116  to histogram $\mathbf{h}_{i,j}$, namely gradients in the square patch
117  $Q_{i,j}$
118  - centered in $C_{i,j}$ as square patch $P_{i,j}$,
119  - with side length $2$.
120  That is because we want to do trilinear interpolation in order to make
121  SIFT robust to small shift in rotation, translation.
122 
123  Therefore, to compute the SIFT descriptor we need to scan all the pixels
124  on a larger circular image patch with radius $r$:
125  */
126  const float r = sqrt(2.f) * l * (N + 1) / 2.f;
127  /*
128  In the above formula, notice:
129  - the factor $\sqrt{2}$ because diagonal corners of the furthest patches
130  $P_{i,j}$ from the center $(x,y)$ must be in the circular patch.
131  - the factor $(N+1)/2$ because we have to include the gradients in larger
132  patches $Q_{i,j}$ for each $P_{i,j}$.
133 
134  I recommend to make a drawing to convince oneself.
135  */
136 
137  // To build the SIFT descriptor, we do the following procedure:
138  // - we work in the image reference frame;
139  // - we scan in the convolved image $G_\sigma$ the position $(x+u, y+v)$
140  // where $(u,v) \in [-r,r]^2$;
141  // - we retrieve its coordinates in the oriented frame of the patch
142  // $P(x,y,\sigma,\theta)$ with inverse transform $T = 1/l R_\theta^T$
143  Matrix2f T;
144  T << cos(theta), sin(theta),
145  -sin(theta), cos(theta);
146  T /= l;
147  // Loop to perform interpolation
148  const int rounded_r = int_round(r);
149  const int rounded_x = int_round(x);
150  const int rounded_y = int_round(y);
151  for (int v = -rounded_r; v <= rounded_r; ++v)
152  {
153  for (int u = -rounded_r; u <= rounded_r; ++u)
154  {
155  // Compute the coordinates in the rescaled and oriented coordinate
156  // frame bound to patch $P(k)$.
157  Vector2f pos{ T*Vector2f(u, v) };
158  // subpixel correction?
159  /*pos.x() -= (x - rounded_x);
160  pos.y() -= (y - rounded_y);*/
161 
162  if ( rounded_x+u < 0 || rounded_x+u >= grad_polar_coords.width() ||
163  rounded_y+v < 0 || rounded_y+v >= grad_polar_coords.height() )
164  continue;
165 
166  // Compute the Gaussian weight which gives more emphasis to gradient
167  // closer to the center.
168  float weight = exp(-pos.squaredNorm()/(2.f*pow(N/2.f, 2)));
169  float mag = grad_polar_coords(rounded_x+u, rounded_y+v)(0);
170  float ori = grad_polar_coords(rounded_x+u, rounded_y+v)(1) - theta;
171  ori = ori < 0.f ? ori+2.f*pi : ori;
172  ori *= float(O)/(2.f*pi);
173 
174  // The coordinate frame is centered in the patch center, thus:
175  // $(x,y)$ is in $[-(N+1)/2, (N+1)/2]^2$.
176  //
177  // Change the coordinate frame so that $(x,y)$ is in $[-1, N]^2$. Thus,
178  // translate by $[ (N-1)/2, (N-1)/2 ]$.
179  pos.array() += N/2.f - 0.5f;
180  if (pos.minCoeff() <= -1.f || pos.maxCoeff() >= static_cast<float>(N))
181  continue;
182  // In the translated coordinate frame, note that for $N=4$ the centers
183  // are now located at:
184  // (0,0) (1,0) (2,0) (3,0)
185  // (0,1) (1,1) (2,1) (3,1)
186  // (0,2) (1,1) (2,2) (3,2)
187  // (0,3) (1,1) (2,3) (3,3)
188  //
189 
190  // Update the SIFT descriptor using trilinear interpolation.
191  accumulate(h, pos, ori, weight, mag);
192  }
193  }
194 
195  h.normalize();
196 
197  h = (h * 512.f).cwiseMin(Matrix<float, Dim, 1>::Ones()*255.f);
198  return h;
199  }

259  private: /* member functions. */
261  void accumulate(descriptor_type& h, const Vector2f& pos, float ori,
262  float weight, float mag) const
263  {
264  // By trilinear interpolation, we mean that in this translated coordinate
265  // frame, a gradient with orientation $\theta$ and located at
266  // $(x,y) \in [-1,N]^2$ contributes to the 4 histograms:
267  // - $\mathbf{h}_{ floor(y) , floor(x) }$
268  // - $\mathbf{h}_{ floor(y) , floor(x)+1}$
269  // - $\mathbf{h}_{ floor(y)+1, floor(x) }$
270  // - $\mathbf{h}_{ floor(y)+1, floor(x)+1}$
271  // In each of these histograms, the following bins are accumulated:
272  // - $\mathbf{h}_{o}$
273  // - $\mathbf{h}_{o+1}$
274  // where $o = floor(\theta * O/ (2*\pi))$
275  //
276  // Note that a gradient at the boundary like $(-1,-1)$ contributes only
277  // to P_{0,0}.
278  float xfrac = pos.x() - floor(pos.x());
279  float yfrac = pos.y() - floor(pos.y());
280  float orifrac = ori - floor(ori);
281  int xi = int(pos.x());
282  int yi = int(pos.y());
283  int orii = int(ori);
284  for (int dy = 0; dy < 2; ++dy)
285  {
286  int y = yi + dy;
287  if (y < 0 || y >= N)
288  continue;
289  float wy = (dy == 0) ? 1.f - yfrac : yfrac;
290  for (int dx = 0; dx < 2; ++dx)
291  {
292  int x = xi+dx;
293  if (x < 0 || x >= N)
294  continue;
295  float wx = (dx == 0) ? 1.f - xfrac : xfrac;
296  for (int dori = 0; dori < 2; ++dori)
297  {
298  int o = (orii + dori) % O;
299  float wo = (dori == 0) ? 1.f - orifrac : orifrac;
300  // Trilinear interpolation:
301  // SIFT(y,x,o) += wy*wx*wo*weight*mag;
302  h[at(y, x, o)] += wy*wx*wo*weight*mag;
303  }
304  }
305  }
306  }

309  void normalize(descriptor_type& h)
310  {
311  // Euclidean normalization.
312  h.normalize();
313  // Clamp histogram bin values $h_i$ to 0.2 for enhanced robustness to
314  // lighting change.
315  h = h.cwiseMin(descriptor_type::Ones()*_max_bin_value);
316  // Renormalize again.
317  h.normalize();
318  }
