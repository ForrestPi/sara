26  template <int N=4, int O=8>
27  class ComputeSIFTDescriptor
28  {
29  public: /* interface. */
30  enum { Dim = N*N*O };
31  typedef Descriptor<float, Dim> SIFT;
33  ComputeSIFTDescriptor(float binScaleUnitLength = 3.f,
34  float maxBinValue = 0.2f)
35  : bin_scale_unit_length_(binScaleUnitLength), max_bin_value_(maxBinValue) {}
37  SIFT operator()(float x, float y, float sigma, float theta,
38  const Image<Vector2f>& gradPolar) const
39  {
40  const float pi = static_cast<float>(M_PI);
41  /*
42  The oriented keypoint is denoted by $k = (x,y,\sigma,\theta)$.
43  SIFT describes keypoint $k$ in a similarity-invariant manner.
44 
45  To do so, we consider a square image patch which:
46  - is centered in $(x,y)$
47  - has an orientation angle $\theta$ w.r.t. the image frame coordinates:
48  => to ensure rotation invariance
49  - has a side length proportional to the scale $\sigma$:
50  => to ensure scale invariance
51  This square patch is denoted by $P(x,y,\sigma,\theta) = P(k)$.
52 
53  The square patch $P(x,y,\sigma,\theta)$ is itself divided into NxN
54  smaller square patches $(P_{i,j})_{1 \leq i \leq N, j \leq j \leq N}$.
55 
56  Notice that we omit the variables $(x,y,\sigma,\theta)$ which the
57  patches $P_{i,j}$ actually depend on.
58 
59  $N$ corresponds to the template argument 'int N' which should be 4 as
60  stated in the paper [Lowe, IJCV 2004]).
61 
62  In the image, each small square patch $P_{i,j}$ has a side length $l$
63  proportional to the scale $\sigma$ of the keypoint, i.e.,
64  $l = \lambda \sigma$.
65  */
66  const float lambda = bin_scale_unit_length_;
67  const float l = lambda*sigma;
68  /*
69  It is important to note that $\lambda$ is some 'universal' constant
70  used for all SIFT descriptors to ensure the scale-invariance of the
71  descriptor.
72  */
73 
74  /*
75  Now in each image square patch $P_{i,j}$, we build a histogram of
76  gradient orientations $\mathbf{h}_{i,j} \in \mathbb{R}^d$, which
77  quantizes the gradient orientations into $O$ principal orientations.
78  $O$ corresponds to the template argument 'int O'.
79 
80  Let us initialize the SIFT descriptor consisting of the NxN histograms
81  $\mathbf{h}_{i,j}$, each in $\mathbf{R}^O$ as follows.
82  */
83  SIFT h(SIFT::Zero());
84 
85  /*
86  In the rescaled and oriented coordinate frame bound to the patch $P(k)$,
87  - keypoint $k$ is located at (0,0)
88  - centers $C_{i,j}$ of patch $P_{i,j}$ are located at
89  $[ -(N+1)/2 + i, -(N+1)/2 + j ]$
90 
91  For example for $N=4$, they are at:
92  (-1.5,-1.5) (-0.5,-1.5) (0.5,-1.5) (1.5,-1.5)
93  (-1.5,-0.5) (-0.5,-0.5) (0.5,-0.5) (1.5,-0.5)
94  (-1.5, 0.5) (-0.5, 0.5) (0.5, 0.5) (1.5, 0.5)
95  (-1.5, 1.5) (-0.5, 1.5) (0.5, 1.5) (1.5, 1.5)
96 
97  Gradients in $[x_i-1, x_i+1] \times [y_i-1, y_i+1]$ contributes
98  to histogram $\mathbf{h}_{i,j}$, namely gradients in the square patch
99  $Q_{i,j}$
100  - centered in $C_{i,j}$ as square patch $P_{i,j}$,
101  - with side length $2$.
102  That is because we want to do trilinear interpolation in order to make
103  SIFT robust to small shift in rotation, translation.
104 
105  Therefore, to compute the SIFT descriptor we need to scan all the pixels
106  on a larger circular image patch with radius $r$:
107  */
108  const float r = sqrt(2.f) * l * (N+1)/2.f;
109  /*
110  In the above formula, notice:
111  - the factor $\sqrt{2}$ because diagonal corners of the furthest patches
112  $P_{i,j}$ from the center $(x,y)$ must be in the circular patch.
113  - the factor $(N+1)/2$ because we have to include the gradients in larger
114  patches $Q_{i,j}$ for each $P_{i,j}$.
115  It is recommended to make a drawing to convince oneself.
116  */
117 
118  // To build the SIFT descriptor, we do the following procedure:
119  // - we work in the image reference frame;
120  // - we scan in the convolved image $G_\sigma$ the position $(x+u, y+v)$
121  // where $(u,v) \in [-r,r]^2$;
122  // - we retrieve its coordinates in the oriented frame of the patch
123  // $P(x,y,\sigma,\theta)$ with inverse transform $T = 1/l R_\theta^T$
124  Matrix2f T;
125  T << cos(theta), sin(theta),
126  -sin(theta), cos(theta);
127  T /= l;
128  // Loop to perform interpolation
129  const int rounded_r = intRound(r);
130  const float rounded_x = intRound(x);
131  const float rounded_y = intRound(y);
132  for (int v = -rounded_r; v <= rounded_r; ++v)
133  {
134  for (int u = -rounded_r; u <= rounded_r; ++u)
135  {
136  // Compute the coordinates in the rescaled and oriented coordinate
137  // frame bound to patch $P(k)$.
138  Vector2f pos( T*Vector2f(u,v) );
139  // subpixel correction?
140  /*pos.x() -= (x - rounded_x);
141  pos.y() -= (y - rounded_y);*/
142 
143  if ( rounded_x+u < 0 || rounded_x+u >= gradPolar.width() ||
144  rounded_y+v < 0 || rounded_y+v >= gradPolar.height() )
145  continue;
146 
147  // Compute the Gaussian weight which gives more emphasis to gradient
148  // closer to the center.
149  float weight = exp(-pos.squaredNorm()/(2.f*pow(N/2.f, 2)));
150  float mag = gradPolar(rounded_x+u, rounded_y+v)(0);
151  float ori = gradPolar(rounded_x+u, rounded_y+v)(1) - theta;
152  ori = ori < 0.f ? ori+2.f*pi : ori;
153  ori *= float(O)/(2.f*pi);
154 
155  // The coordinate frame is centered in the patch center, thus:
156  // $(x,y)$ is in $[-(N+1)/2, (N+1)/2]^2$.
157  //
158  // Change the coordinate frame so that $(x,y)$ is in $[-1, N]^2$. Thus,
159  // translate by $[ (N-1)/2, (N-1)/2 ]$.
160  pos.array() += N/2.f - 0.5f;
161  if (pos.minCoeff() <= -1.f || pos.maxCoeff() >= static_cast<float>(N))
162  continue;
163  // In the translated coordinate frame, note that for $N=4$ the centers
164  // are now located at:
165  // (0,0) (1,0) (2,0) (3,0)
166  // (0,1) (1,1) (2,1) (3,1)
167  // (0,2) (1,1) (2,2) (3,2)
168  // (0,3) (1,1) (2,3) (3,3)
169  //
170 
171  // Update the SIFT descriptor using trilinear interpolation.
172  accumulate(h, pos, ori, weight, mag);
173  }
174  }
175 
176  h.normalize();
177 
178  h = (h * 512.f).cwiseMin(Matrix<float, Dim, 1>::Ones()*255.f);
179  return h;
180  }

226  private: /* member functions. */
228  void accumulate(SIFT& h, const Vector2f& pos, float ori,
229  float weight, float mag) const
230  {
231  // By trilinear interpolation, we mean that in this translated coordinate
232  // frame, a gradient with orientation $\theta$ and located at
233  // $(x,y) \in [-1,N]^2$ contributes to the 4 histograms:
234  // - $\mathbf{h}_{ floor(y) , floor(x) }$
235  // - $\mathbf{h}_{ floor(y) , floor(x)+1}$
236  // - $\mathbf{h}_{ floor(y)+1, floor(x) }$
237  // - $\mathbf{h}_{ floor(y)+1, floor(x)+1}$
238  // In each of these histograms, the following bins are accumulated:
239  // - $\mathbf{h}_{o}$
240  // - $\mathbf{h}_{o+1}$
241  // where $o = floor(\theta * O/ (2*\pi))$
242  //
243  // Note that a gradient at the boundary like $(-1,-1)$ contributes only
244  // to P_{0,0}.
245  float xfrac = pos.x() - floor(pos.x());
246  float yfrac = pos.y() - floor(pos.y());
247  float orifrac = ori - floor(ori);
248  int xi = int(pos.x());
249  int yi = int(pos.y());
250  int orii = int(ori);
251  for (int dy = 0; dy < 2; ++dy)
252  {
253  int y = yi+dy;
254  if (y < 0 || y >= N)
255  continue;
256  float wy = (dy == 0) ? 1.f-yfrac : yfrac;
257  for (int dx = 0; dx < 2; ++dx)
258  {
259  int x = xi+dx;
260  if (x < 0 || x >= N)
261  continue;
262  float wx = (dx == 0) ? 1.f-xfrac : xfrac;
263  for (int dori = 0; dori < 2; ++dori)
264  {
265  int o = (orii+dori)%O;
266  float wo = (dori == 0) ? 1.f-orifrac : orifrac;
267  // Trilinear interpolation:
268  // SIFT(y,x,o) += wy*wx*wo*weight*mag;
269  h[at(y,x,o)] += wy*wx*wo*weight*mag;
270  }
271  }
272  }
273  }
275  void normalize(SIFT& h)
276  {
277  // Euclidean normalization.
278  h.normalize();
279  // Clamp histogram bin values $h_i$ to 0.2 for enhanced robustness to
280  // lighting change.
281  h = h.cwiseMin(SIFT::Ones()*max_bin_value_);
282  // Renormalize again.
283  h.normalize();
284  }
286  inline int at(int i, int j, int o) const
287  { return N*O*i + j*O + o; }
