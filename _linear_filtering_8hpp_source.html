44  void convolveArray(T *signal,
45  const typename ColorTraits<T>::ChannelType *kernel,
46  int signalSz, int kernelSz)
47  {
48  T *bp;
49  T *b = signal;
50  const typename ColorTraits<T>::ChannelType *kp;
51 
52  for (int i = 0; i < signalSz; ++i)
53  {
54  bp = b;
55  kp = kernel;
56 
57  T sum(ColorTraits<T>::zero());
58  for (int j = 0; j < kernelSz; j++)
59  sum += *bp++ * *kp++;
60 
61  *b++ = sum;
62  }
63  }

76  template <typename T>
77  void applyRowBasedFilter(Image<T>& dst, const Image<T>& src,
78  const typename ColorTraits<T>::ChannelType *kernel,
79  int kernelSize)
80  {
81  const int w = src.width();
82  const int h = src.height();
83  const int halfSize = kernelSize/2;
84  Image<T> buffer(w+halfSize*2,1);
85 
86  // Resize if necessary.
87  if (dst.sizes() != src.sizes())
88  dst.resize(w,h);
89 
90 
91  for (int y = 0; y < h; ++y) {
92  // Copy to work array and add padding.
93  for (int x = 0; x < halfSize; ++x)
94  buffer(x,0) = src(0,y);
95  for (int x = 0; x < w; ++x)
96  buffer(halfSize+x,0) = src(x,y);
97  for (int x = 0; x < halfSize; ++x)
98  buffer(w+halfSize+x,0) = src(w-1,y);
99 
100  // Compute the value by convolution
101  for (int x = 0; x < w; ++x) {
102  dst(x,y) = ColorTraits<T>::zero();
103  for (int k = 0; k < kernelSize; ++k)
104  dst(x,y) += kernel[k]*buffer(x+k,0);
105  }
106  }
107  }

162  const typename ColorTraits<T>::ChannelType *kernel,
163  int kernelSize)

168  T *buffer = new T[w+halfSize*2];

170  {
171  // Copy to work array and add padding.
172  for (int x = 0; x < halfSize; ++x)
173  buffer[x] = src(0,y);
174  for (int x = 0; x < w; ++x)

243  template <typename T>
244  void applyGaussianFilter(Image<T>& dst, const Image<T>& src,
245  typename ColorTraits<T>::ChannelType sigma,
246  typename ColorTraits<T>::ChannelType gaussTruncate =
247  typename ColorTraits<T>::ChannelType(4))
248  {
249  DO_STATIC_ASSERT(
250  !std::numeric_limits<typename ColorTraits<T>::ChannelType >::is_integer,
251  CHANNEL_TYPE_MUST_NOT_BE_INTEGRAL );
252 
253  typedef typename ColorTraits<T>::ChannelType S;
254 
255  // Compute the size of the Gaussian kernel.
256  int kSize = int(S(2) * gaussTruncate * sigma + S(1));
257  // Make sure the Gaussian kernel is at least of size 3 and is of odd size.
258  kSize = std::max(3, kSize);
259  if (kSize % 2 == 0)
260  ++kSize;
261 
262  // Create the 1D Gaussian kernel.
263  S *kernel = new S[kSize];
264  S sum(0);
265 
266  // Compute the value of the Gaussian and the normalizing factor.
267  for (int i = 0; i < kSize; ++i)
268  {
269  S x = S(i - kSize/2);
270  kernel[i] = exp(-x*x/(S(2)*sigma*sigma));
271  sum += kernel[i];
272  }
273 
274  // Normalize the kernel.
275  for (int i = 0; i < kSize; ++i)
276  kernel[i] /= sum;
277 
278  applyFastRowBasedFilter(dst, src, &kernel[0], kSize);
279  applyFastColumnBasedFilter(dst, dst, &kernel[0], kSize);
280 
281  delete[] kernel;
282  }

286  {

288  S meanKernel[] = { S(1), S(2), S(1) };
289  S diffKernel[] = { S(1), S(0), S(-1) };

339  template <typename T>
340  void apply2DNonSeparableFilter(Image<T>& dst, const Image<T>& src,
341  const typename ColorTraits<T>::ChannelType *kernel,
342  int kWidth, int kHeight)
343  {
344  typedef typename Image<T>::coords_type Coords;
345 
346  const int hkWidth = kWidth/2;
347  const int hkHeight = kHeight/2;
348  const int w = src.width();
349  const int h = src.height();
350 
351  Coords sizes(src.sizes()+Coords(hkWidth*2, hkHeight*2));
352  Image<T> work(sizes);
353  const int workw = work.width();
354  const int workh = work.height();
355 
356  for (int y = 0; y < workh; ++y) {
357  for (int x = 0; x < workw; ++x) {
358  // North-West
359  if (x < hkWidth && y < hkHeight)
360  work(x,y) = src(0,0);
361  // West
362  if (x < hkWidth && hkHeight <= y && y < h+hkHeight)
363  work(x,y) = src(0,y-hkHeight);
364  // South-West
365  if (x < hkWidth && y >= h+hkHeight)
366  work(x,y) = src(0,h-1);
367  // North
368  if (hkWidth <= x && x < w+hkWidth && y < hkHeight)
369  work(x,y) = src(x-hkWidth,0);
370  // South
371  if (hkWidth <= x && x < w+hkWidth && y >= h+hkHeight)
372  work(x,y) = src(x-hkWidth,h-1);
373  // North-East
374  if (x >= w+hkWidth && y >= h+hkHeight)
375  work(x,y) = src(w-1,0);
376  // East
377  if (x >= w+hkWidth && hkHeight <= y && y < h+hkHeight)
378  work(x,y) = src(w-1,y-hkHeight);
379  // South-East
380  if (x >= w+hkWidth && y >= h+hkHeight)
381  work(x,y) = src(w-1,h-1);
382  // Middle
383  if (hkWidth <= x && x < w+hkWidth && hkHeight <= y && y < h+hkHeight)
384  work(x,y) = src(x-hkWidth,y-hkHeight);
385  }
386  }
387 
388  // Resize if necessary.
389  if (dst.sizes() != src.sizes())
390  dst.resize(src.sizes());
391 
392  // Convolve
393  for (int y = 0; y < h; ++y) {
394  for (int x = 0; x < w; ++x) {
395  T val(ColorTraits<T>::zero());
396  for (int yy = 0; yy < kHeight; ++yy)
397  for (int xx = 0; xx < kWidth; ++xx)
398  val += work(x+xx, y+yy)
399  * kernel[yy*kWidth + xx];
400  dst(x,y) = val;
401  }
402  }
403  }
