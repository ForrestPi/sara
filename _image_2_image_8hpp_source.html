151  -> Image<decltype(op(std::declval<pixel_type>())), Dimension>
152  {
153  using PixelType = decltype(op(std::declval<pixel_type>()));
154  Image<PixelType, Dimension> dst{ this->sizes() };
155  auto src_pixel = this->begin();
156  auto dst_pixel = dst.begin();
