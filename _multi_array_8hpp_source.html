204  const self_type& operator=(const self_type& M)
205  {
206  // We must be careful when (M == *this)!
207  if (!is_wrapped_data_ && begin_ != M.data())
208  {
209  delete[] begin_;
210  begin_ = 0;
211  end_ = 0;
212  }
213 
214  // Allocate new data.
215  T *newData = new T[M.size()];
216  sizes_ = M.sizes();
217  strides_ = M.strides();
218  is_wrapped_data_ = false;
219 
220  if (newData)
221  std::copy(M.data(), M.data() + M.size(), newData);
222 
223  // We must be careful when (M == *this)!
224  if (begin_ == M.data())
225  delete[] begin_;
226  begin_ = newData;
227  end_ = begin_ + M.size();
228 
229  return *this;
230  }
233  template <typename T2>
