428  inline void copy(const Tree& other)
429  {
430  clear();
431 
432  if (other.empty())
433  return;
434 
435  // Mapping between source node and destination nodes.
436  std::map<const Node *, Node *> src_to_dst;
437 
438  // Initialize the copy.
439  set_root(*other.begin());
440  src_to_dst[other.begin()] = begin();
441 
442  // Loop.
443  auto src_node = ++other.depth_first_begin();
444  auto src_node_end = other.depth_first_end();
445  for ( ; src_node != src_node_end; ++src_node)
446  {
447  // Parent of the source node.
448  auto src_parent_node = src_node.parent();
449  // Parent of the destination node.
450  auto dst_parent_node = src_to_dst[src_parent_node];
451 
452  // Create the new node.
453  auto dst_node = new Node(*src_node);
454 
455  // Connect the parent and the new child.
456  dst_parent_node->append_child(dst_node);
457 
458  // Update the mapping.
459  src_to_dst[src_node] = dst_node;
