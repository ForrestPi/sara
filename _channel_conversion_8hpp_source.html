40  template <typename Int, typename Float>
41  inline Float to_normalized_float_channel(Int src)
42  {
43  static_assert(
44  std::numeric_limits<Int>::is_integer,
45  "Channel conversion must be from integer type to floating point type");
46 
47  using std::numeric_limits;
48  const auto float_min = static_cast<Float>(numeric_limits<Int>::min());
49  const auto float_max = static_cast<Float>(numeric_limits<Int>::max());
50  const auto float_range = float_max - float_min;
51  return (static_cast<Float>(src) - float_min) / float_range;
52  }

55  template <typename Int, typename Float>
56  inline Int to_rescaled_integral_channel(Float src)
57  {
58  static_assert(
59  std::numeric_limits<Int>::is_integer,
60  "Channel conversion must be from floating point type to integer type");
61 
62  using std::numeric_limits;
63  auto float_min = static_cast<Float>(numeric_limits<Int>::min());
64  auto float_max = static_cast<Float>(numeric_limits<Int>::max());
65  auto float_range = float_max - float_min;
66  src = float_min + src * float_range;
67 
68  const auto delta_max = std::abs(src - float_max) / float_range;
69  const auto delta_min = std::abs(src - float_min) / float_range;
70  const auto eps = sizeof(Float) == 4 ?
71  Float(1e-5) : // i.e., if 'Float' == 'float'.
72  Float(1e-9); // i.e., if 'Float' == 'double'.
73 
74  if (delta_max <= eps)
75  return std::numeric_limits<Int>::max();
76 
77  if (delta_min <= eps)
78  return std::numeric_limits<Int>::min();
79 
80  return static_cast<Int>(floor(src + 0.5));
81  }
