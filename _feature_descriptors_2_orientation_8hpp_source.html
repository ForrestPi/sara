46  Image<Matrix<T,2,1>> gradient_polar_coordinates(const Image<T>& f)
47  {
48  Image<Matrix<T, 2, 1>> nabla_f{ gradient(f) };
49  for (auto it = nabla_f.begin(); it != nabla_f.end(); ++it)
50  {
51  auto r = 2*it->norm();
52  auto theta = atan2(it->y(), it->x());
53  *it << r, theta;
54  }
55  return nabla_f;

84  template <typename T, int N>
85  void compute_orientation_histogram(Array<T, N, 1>& orientation_histogram,
86  const Image<Matrix<T,2,1>>& grad_polar_coords,
87  T x, T y, T s,
88  T patch_truncation_factor = T(3),
89  T blur_factor = T(1.5))
90  {
91  // Weighted histogram of gradients.
92  orientation_histogram.setZero();
93 
94  // Rounding of the coordinates.
95  auto rounded_x = int_round(x);
96  auto rounded_y = int_round(y);
97 
98  // std deviation of the gaussian weight (cf. [Lowe, IJCV 2004])
99  auto sigma = s*blur_factor;
100 
101  // Patch radius on which the histogram of gradients is performed.
102  auto patch_radius = int_round(sigma*patch_truncation_factor);
103 
104  // Accumulate the histogram of orientations.
105  for (auto v = -patch_radius; v <= patch_radius; ++v)
106  {
107  for (auto u = -patch_radius; u <= patch_radius; ++u)
108  {
109  if (rounded_x + u < 0 || rounded_x + u >= grad_polar_coords.width() ||
110  rounded_y + v < 0 || rounded_y + v >= grad_polar_coords.height())
111  continue;
112 
113  auto mag = grad_polar_coords(rounded_x+u, rounded_y+v)(0);
114  auto ori = grad_polar_coords(rounded_x+u, rounded_y+v)(1);
115 
116  // ori is in \f$]-\pi, \pi]\f$, so translate ori by \f$2*\pi\f$ if it is
117  // negative.
118  ori = ori < 0 ? ori + T(2.*M_PI) : ori;
119  auto bin_index = int(floor(ori / T(2 * M_PI) * N));
120  bin_index %= N;
121 
122  // Give more emphasis to gradient orientations that lie closer to the
123  // keypoint location.
124  auto weight = exp(-(u*u + v*v) / (T(2)*sigma*sigma));
125  // Also give more emphasis to gradient with large magnitude.
126  orientation_histogram(bin_index) += weight*mag;
127  }
128  }
129  }

137  template <typename T, int N>
138  void lowe_smooth_histogram(Array<T, N, 1>& orientation_histogram)
139  {
140  for (int iter = 0; iter < 6; ++iter)
141  {
142  float first = orientation_histogram(0);
143  float prev = orientation_histogram(N-1);
144  for (int i = 0; i < N-1; ++i)
145  {
146  float val = (prev+orientation_histogram(i)+orientation_histogram(i+1))/3.f;
147  prev = orientation_histogram(i);
148  orientation_histogram(i) = val;
149  }
150  orientation_histogram(N-1) = (prev+orientation_histogram(N-1)+first)/3.f;
151  }

163  template <typename T, int N>
164  std::vector<int> find_peaks(const Array<T, N, 1>& orientation_histogram,
165  T peak_ratio_thres = T(0.8))
166  {
167  T max = orientation_histogram.maxCoeff();
168  std::vector<int> orientation_peaks;
169  orientation_peaks.reserve(N);
170  for (int i = 0; i < N; ++i)
171  if ( orientation_histogram(i) >= peak_ratio_thres*max &&
172  orientation_histogram(i) > orientation_histogram((i - 1 + N) % N) &&
173  orientation_histogram(i) > orientation_histogram((i + 1) % N) )
174  orientation_peaks.push_back(i);
175  return orientation_peaks;
176  }

182  template <typename T, int N>
183  T refine_peak(const Array<T, N, 1>& orientation_histogram, int i)
184  {
185  T y0 = orientation_histogram( (i-1+N) % N );
186  T y1 = orientation_histogram( i );
187  T y2 = orientation_histogram( (i+1) % N );
188  // Denote the orientation histogram function by \f$f\f$.
189  // perform a 2nd-order Taylor approximation:
190  // \f$f(x+h) = f(x) + f'(x)h + f''(x) h^2/2\f$
191  // We approximate \f$f'\f$ and \f$f''\f$ by finite difference.
192  T fprime = (y2-y0) / 2.f;
193  T fsecond = y0 - 2.f*y1 + y2;
194  // Maximize w.r.t. to \f$h\f$, derive the expression.
195  // Thus \f$h = -f'(x)/f''(x)\f$.
196  T h = -fprime / fsecond;
197  // Add the offset \f$h\f$ to get the refined orientation value.
198  // Note that we also add the 0.5f offset, because samples are assumed taken
199  // on the middle of the interval \f$[i, i+1)\f$.
200  return T(i)+T(0.5)+h;
201  }
