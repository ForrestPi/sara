134  QRectF computeBoundingRect(const Feature& f)
135  {
136  typedef typename Feature::Matrix2x2 Matrix2x2;
137  Eigen::JacobiSVD<Matrix2x2> svd(f.scaleMatrix());
138  const qreal a = qreal(1)/std::sqrt(svd.singularValues()(0));
139  const qreal b = qreal(1)/std::sqrt(svd.singularValues()(1));
140  return QRectF(-a, -b, 2*a, 2*b);
141  }

175  void adjust()
176  {
177  if (!source_ || !target_)
178  return;
179 
180  QLineF line(mapFromItem(source_, 0, 0),
181  mapFromItem(target_, 0, 0));
182  qreal length = line.length();
183 
184  prepareGeometryChange();
185 
186  if (length > qreal(20.))
187  {
188  QPointF edgeOffset( (line.dx()*5) / length,
189  (line.dy()*5) / length);
190  sourcePoint_ = line.p1() + edgeOffset;
191  targetPoint_ = line.p2() - edgeOffset;
192  }
193  else
194  sourcePoint_ = targetPoint_ = line.p1();
195  }

212  QRectF boundingRect() const
213  {
214  if (!source_ || !target_)
215  return QRectF();
216

218  qreal extra = (penWidth + arrowSize_) / 2.0;
219 
220  return QRectF(sourcePoint_,
221  QSizeF(targetPoint_.x() - sourcePoint_.x(),
222  targetPoint_.y() - sourcePoint_.y()))
223  .normalized()
224  .adjusted(-extra, -extra, extra, extra);

227  void paint(QPainter* painter,
228  const QStyleOptionGraphicsItem* option,
229  QWidget* widget)
230 
231  {
232  if (!source_ || !target_)
233  return;
234 
235  QLineF line(sourcePoint_, targetPoint_);
236  if (qFuzzyCompare(line.length(), qreal(0.)))
237  return;
238 
239  // Draw the line itself
240  painter->setPen(QPen(currentColor_, 2,
241  Qt::SolidLine, Qt::RoundCap,
242  Qt::RoundJoin));
243  painter->drawLine(line);
244 
245  const qreal Pi = pi<qreal>();
246 
247  // Draw the arrows
248  double angle = std::atan2(line.dy(), line.dx());
249 
250  QPointF sourceArrowP1 = sourcePoint_
251  + QPointF(cos(angle+Pi/12)*arrowSize_,
252  sin(angle+Pi/12)*arrowSize_);
253  QPointF sourceArrowP2 = sourcePoint_
254  + QPointF(cos(angle-Pi/12)*arrowSize_,
255  sin(angle-Pi/12)*arrowSize_);
256  QPointF targetArrowP1 = targetPoint_
257  + QPointF(cos(angle+Pi-Pi/12)*arrowSize_,
258  sin(angle+Pi-Pi/12)*arrowSize_);
259  QPointF targetArrowP2 = targetPoint_
260  + QPointF(cos(angle+Pi+Pi/12)*arrowSize_,
261  sin(angle+Pi+Pi/12)*arrowSize_);
262 
263  painter->drawPolygon(QPolygonF() << line.p1() <<
264  sourceArrowP1 << sourceArrowP2);
265  painter->drawPolygon(QPolygonF() << line.p2() <<
266  targetArrowP1 << targetArrowP2);
267  }
