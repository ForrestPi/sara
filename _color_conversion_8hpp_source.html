25  template <typename T, int N>
26  inline void rgb_to_gray(const Matrix<T, N, 1>& rgb, T& gray)
27  {
28  static_assert(
29  !std::numeric_limits<T>::is_integer,
30  "Conversion from rgb to gray is supported only for floating point type");
31 
32  gray = T(0.2125)*rgb[0] + T(0.7154)*rgb[1] + T(0.0721)*rgb[2];
33  }

43  template <typename T>
44  inline void rgb_to_yuv(const Matrix<T, 3, 1>& rgb, Matrix<T, 3, 1>& yuv)
45  {
46  static_assert(
47  !std::numeric_limits<T>::is_integer,
48  "Conversion from gray to RGB is supported only for floating point type");
49 
50  yuv[0] = T(0.299)*rgb[0] + T(0.587)*rgb[1] + T(0.114)*rgb[2];
51  yuv[1] = T(0.492)*(rgb[2] - yuv[0]);
52  yuv[2] = T(0.877)*(rgb[0] - yuv[0]);
53  }

56  template <typename T>
57  inline void yuv_to_rgb(const Matrix<T, 3, 1>& yuv, Matrix<T, 3, 1>& rgb)
58  {
59  static_assert(
60  !std::numeric_limits<T>::is_integer,
61  "Conversion from gray to RGB is supported only for floating point type");
62 
63  rgb[0] = yuv[0] + T(1.13983)*yuv[2];
64  rgb[1] = yuv[0] - T(0.39465)*yuv[1] - T(0.58060)*yuv[2];
65  rgb[2] = yuv[0] + T(2.03211)*yuv[1];
66  }
